# ocr-service/main.py

import cv2

app = FastAPI(title="律所OCR服务", version="1.0.0")

# 初始化PaddleOCR（惰性加载）
_ocr_engine = None

def get_ocr_engine():
    """获取OCR引擎（单例模式）"""
    global _ocr_engine
    if _ocr_engine is None:
        model_dir = os.getenv('OCR_MODEL_DIR', '/app/models')

        )
    return _ocr_engine

# 银行回单解析器
class BankStatementParser:
    """银行回单专用解析器"""
    
    @staticmethod
    def parse_amount(text: str) -> Dict[str, Any]:
        """解析金额"""
        amount_patterns = [
            r'金额[：:]?\s*([￥¥]?\s*\d{1,3}(?:,\d{3})*(?:\.\d{2})?)',
            r'人民币[：:]?\s*(\d{1,3}(?:,\d{3})*(?:\.\d{2})?)\s*元',
            r'小写[：:]?\s*([￥¥]?\s*\d{1,3}(?:,\d{3})*(?:\.\d{2})?)',
            r'¥\s*(\d{1,3}(?:,\d{3})*(?:\.\d{2})?)',
            r'￥\s*(\d{1,3}(?:,\d{3})*(?:\.\d{2})?)'
        ]
        
        for pattern in amount_patterns:
            match = re.search(pattern, text, re.IGNORECASE)
            if match:
                amount_str = match.group(1).replace(',', '').replace('¥', '').replace('￥', '').strip()
                try:
                    amount = float(amount_str)
                    return {'amount': amount, 'text': amount_str, 'pattern': pattern}
                except:
                    continue
        
        return {'amount': None, 'text': '', 'pattern': None}
    
    @staticmethod
    def parse_date(text: str) -> Dict[str, Any]:
        """解析日期"""
        date_patterns = [
            r'(\d{4})年(\d{1,2})月(\d{1,2})日',
            r'(\d{4})-(\d{1,2})-(\d{1,2})',
            r'(\d{4})/(\d{1,2})/(\d{1,2})',
            r'(\d{4})\.(\d{1,2})\.(\d{1,2})',
            r'日期[：:]?\s*(\d{4}[-/年]\d{1,2}[-/月]\d{1,2}日?)'
        ]
        
        for pattern in date_patterns:
            match = re.search(pattern, text)
            if match:
                try:
                    if '年' in pattern:
                        year, month, day = match.groups()
                    else:
                        year, month, day = match.groups()
                    
                    # 格式化日期
                    month = month.zfill(2)
                    day = day.zfill(2)
                    date_str = f"{year}-{month}-{day}"
                    
                    return {'date': date_str, 'text': match.group(), 'pattern': pattern}
                except:
                    continue
        
        return {'date': None, 'text': '', 'pattern': None}
    
    @staticmethod
    def parse_payer(text: str) -> Dict[str, Any]:
        """解析付款方"""
        payer_patterns = [
            r'付款人[：:]?\s*([\u4e00-\u9fa5a-zA-Z0-9]{2,})',
            r'付款账户[：:]?\s*([\u4e00-\u9fa5a-zA-Z0-9]{2,})',
            r'汇款人[：:]?\s*([\u4e00-\u9fa5a-zA-Z0-9]{2,})',
            r'客户名称[：:]?\s*([\u4e00-\u9fa5a-zA-Z0-9]{2,})',
            r'名称[：:]?\s*([\u4e00-\u9fa5a-zA-Z0-9]{2,})'
        ]
        
        for pattern in payer_patterns:
            match = re.search(pattern, text)
            if match:
                payer = match.group(1).strip()
                # 清理常见噪声
                payer = re.sub(r'[【】\[\]\(\)（）]', '', payer)
                return {'payer': payer, 'text': match.group(), 'pattern': pattern}
        
        return {'payer': None, 'text': '', 'pattern': None}
    
    @staticmethod
    def parse_bank_account(text: str) -> Dict[str, Any]:
        """解析银行账号"""
        account_patterns = [
            r'账号[：:]?\s*([0-9]{10,20})',
            r'账户[：:]?\s*([0-9]{10,20})',
            r'卡号[：:]?\s*([0-9]{10,20})',
            r'([0-9]{16,19})(?=\s|$)'  # 银行卡号
        ]
        
        for pattern in account_patterns:
            match = re.search(pattern, text)
            if match:
                account = match.group(1).strip()
                return {'account': account, 'text': match.group(), 'pattern': pattern}
        
        return {'account': None, 'text': '', 'pattern': None}

@app.post("/api/ocr/recognize")
async def recognize_text(
    file: UploadFile = File(...),
    doc_type: str = "auto"
):
    """OCR识别主接口"""
    request_id = str(uuid.uuid4())
    
    try:
        # 读取文件
        contents = await file.read()
        
        # 保存临时文件
        temp_path = f"/tmp/{request_id}_{file.filename}"
        with open(temp_path, 'wb') as f:
            f.write(contents)
        
        # 执行OCR识别
        start_time = time.time()
        
        # 根据文件类型处理
        if file.filename.lower().endswith('.pdf'):
            # PDF处理（简化版，实际需要安装pdf2image）
            # 这里先假设是图片
            image = cv2.imdecode(np.frombuffer(contents, np.uint8), cv2.IMREAD_COLOR)
        else:
            # 图片处理
            image = cv2.imdecode(np.frombuffer(contents, np.uint8), cv2.IMREAD_COLOR)
        
        if image is None:
            raise HTTPException(status_code=400, detail="无法读取图片文件")
        
        # OCR识别
        ocr_engine = get_ocr_engine()
        result = ocr_engine.ocr(image, cls=True)
        
        # 提取文本
        texts = []
        if result and len(result) > 0:
            for line in result[0]:
                if line and len(line) >= 2:
                    _, (text, confidence) = line
                    texts.append({
                        'text': text,
                        'confidence': float(confidence)
                    })
        
        processing_time = time.time() - start_time
        
        # 结构化解析
        full_text = ' '.join([item['text'] for item in texts])
        structured_data = {}
        
        # 根据文档类型进行解析
        if doc_type == 'bank_statement' or '银行' in full_text or '金额' in full_text:
            parser = BankStatementParser()
            structured_data = {
                'document_type': 'bank_statement',
                'amount': parser.parse_amount(full_text),
                'date': parser.parse_date(full_text),
                'payer': parser.parse_payer(full_text),
                'account': parser.parse_bank_account(full_text),
                'confidence': sum([t['confidence'] for t in texts]) / len(texts) if texts else 0
            }
        
        # 清理临时文件
        if os.path.exists(temp_path):
            os.remove(temp_path)
        
        # 缓存结果
        result_data = {
            'request_id': request_id,
            'texts': texts,
            'structured_data': structured_data,
            'processing_time': processing_time,
            'file_name': file.filename,
            'timestamp': datetime.now().isoformat()
        }
        
        redis_client.setex(
            f"ocr_result:{request_id}",
            3600,  # 1小时过期
            json.dumps(result_data, ensure_ascii=False)
        )
        
        return JSONResponse(content={
            'success': True,
            'code': '200',
            'message': '识别成功',
            'data': result_data,
            'timestamp': datetime.now().isoformat()
        })
        
    except Exception as e:
        return JSONResponse(
            status_code=500,
            content={
                'success': False,
                'code': '500',
                'message': f'识别失败: {str(e)}',
                'data': None,
                'timestamp': datetime.now().isoformat()
            }
        )

#!/bin/bash
# backup/backup.sh - 完整备份脚本

set -e

# 配置参数
BACKUP_DIR="/backup"
LOG_DIR="/backup/logs"
RETENTION_DAYS=30
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
LOG_FILE="${LOG_DIR}/backup_${TIMESTAMP}.log"

# 创建目录
mkdir -p ${BACKUP_DIR}
mkdir -p ${LOG_DIR}
mkdir -p ${BACKUP_DIR}/daily
mkdir -p ${BACKUP_DIR}/weekly
mkdir -p ${BACKUP_DIR}/monthly

# 日志函数
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a ${LOG_FILE}
}

# 错误处理
handle_error() {
    log "备份失败: $1"
    # 发送告警（可选）
    # curl -X POST https://your-webhook-url -d "{\"text\":\"备份失败: $1\"}"
    exit 1
}

# 开始备份
log "开始备份流程..."

# 1. PostgreSQL数据库备份
log "备份PostgreSQL数据库..."
docker exec law-postgres pg_dump -U law_admin --format=custom law_firm > ${BACKUP_DIR}/daily/db_${TIMESTAMP}.dump 2>>${LOG_FILE} || handle_error "数据库备份失败"
log "数据库备份完成，大小: $(du -h ${BACKUP_DIR}/daily/db_${TIMESTAMP}.dump | cut -f1)"

# 2. MinIO数据备份
log "备份MinIO数据..."
docker run --rm \
    -v minio_data:/data \
    -v ${BACKUP_DIR}/daily:/backup \
    alpine tar czf /backup/minio_${TIMESTAMP}.tar.gz /data 2>>${LOG_FILE} || handle_error "MinIO备份失败"
log "MinIO备份完成，大小: $(du -h ${BACKUP_DIR}/daily/minio_${TIMESTAMP}.tar.gz | cut -f1)"

# 3. Redis数据备份（RDB文件）
log "备份Redis数据..."
docker exec law-redis redis-cli --rdb /data/dump.rdb 2>>${LOG_FILE} || handle_error "Redis备份失败"
docker cp law-redis:/data/dump.rdb ${BACKUP_DIR}/daily/redis_${TIMESTAMP}.rdb 2>>${LOG_FILE} || handle_error "Redis文件拷贝失败"
log "Redis备份完成，大小: $(du -h ${BACKUP_DIR}/daily/redis_${TIMESTAMP}.rdb | cut -f1)"

# 4. 应用配置备份
log "备份应用配置..."
tar czf ${BACKUP_DIR}/daily/config_${TIMESTAMP}.tar.gz \
    ./docker-compose.yml \
    ./.env \
    ./nginx/ \
    ./init-db/ \
    ./backup/backup.sh \
    ./backup/restore.sh 2>>${LOG_FILE} || handle_error "配置备份失败"
log "配置备份完成"

# 5. 每周全量备份（如果是周日）
if [ $(date +%u) -eq 7 ]; then
    log "执行每周全量备份..."
    cp ${BACKUP_DIR}/daily/db_${TIMESTAMP}.dump ${BACKUP_DIR}/weekly/
    cp ${BACKUP_DIR}/daily/minio_${TIMESTAMP}.tar.gz ${BACKUP_DIR}/weekly/
    cp ${BACKUP_DIR}/daily/redis_${TIMESTAMP}.rdb ${BACKUP_DIR}/weekly/
    cp ${BACKUP_DIR}/daily/config_${TIMESTAMP}.tar.gz ${BACKUP_DIR}/weekly/
fi

# 6. 每月归档备份（如果是当月1号）
if [ $(date +%d) -eq 1 ]; then
    log "执行每月归档备份..."
    cp ${BACKUP_DIR}/daily/db_${TIMESTAMP}.dump ${BACKUP_DIR}/monthly/
    cp ${BACKUP_DIR}/daily/minio_${TIMESTAMP}.tar.gz ${BACKUP_DIR}/monthly/
    cp ${BACKUP_DIR}/daily/redis_${TIMESTAMP}.rdb ${BACKUP_DIR}/monthly/
    cp ${BACKUP_DIR}/daily/config_${TIMESTAMP}.tar.gz ${BACKUP_DIR}/monthly/
fi

# 7. 清理旧备份
log "清理过期备份文件（保留${RETENTION_DAYS}天）..."
find ${BACKUP_DIR}/daily -type f -mtime +${RETENTION_DAYS} -delete 2>>${LOG_FILE}
find ${BACKUP_DIR}/weekly -type f -mtime +90 -delete 2>>${LOG_FILE}
find ${BACKUP_DIR}/monthly -type f -mtime +365 -delete 2>>${LOG_FILE}
find ${LOG_DIR} -type f -mtime +30 -delete 2>>${LOG_FILE}

# 8. 备份验证
log "验证备份文件..."
check_files=(
    "${BACKUP_DIR}/daily/db_${TIMESTAMP}.dump"
    "${BACKUP_DIR}/daily/minio_${TIMESTAMP}.tar.gz"
    "${BACKUP_DIR}/daily/redis_${TIMESTAMP}.rdb"
    "${BACKUP_DIR}/daily/config_${TIMESTAMP}.tar.gz"
)

for file in "${check_files[@]}"; do
    if [ ! -f "$file" ] || [ ! -s "$file" ]; then
        handle_error "备份文件验证失败: $file"
    fi
done

# 9. 生成备份报告
total_size=$(du -sh ${BACKUP_DIR}/daily | cut -f1)
log "备份完成!"
log "备份时间: ${TIMESTAMP}"
log "备份位置: ${BACKUP_DIR}"
log "总大小: ${total_size}"
log "日志文件: ${LOG_FILE}"

# 10. 发送成功通知（可选）
# curl -X POST https://your-webhook-url -d "{\"text\":\"备份成功: ${TIMESTAMP}\"}"

exit 0

set -e

# 配置参数
BACKUP_DIR="/backup"
RESTORE_DATE="$1"  # 格式：YYYYMMDD_HHMMSS

if [ -z "$RESTORE_DATE" ]; then
    echo "用法: $0 [备份日期_时间]"
    echo "可用备份:"
    ls -1 ${BACKUP_DIR}/daily/db_*.dump | sed 's/.*db_\(.*\)\.dump/\1/' | sort -r
    exit 1

# 检查备份文件是否存在
BACKUP_FILES=(
    "${BACKUP_DIR}/daily/db_${RESTORE_DATE}.dump"
    "${BACKUP_DIR}/daily/minio_${RESTORE_DATE}.tar.gz"
    "${BACKUP_DIR}/daily/redis_${RESTORE_DATE}.rdb"
    "${BACKUP_DIR}/daily/config_${RESTORE_DATE}.tar.gz"
)

for file in "${BACKUP_FILES[@]}"; do
    if [ ! -f "$file" ]; then
        echo "错误: 备份文件不存在: $file"
        exit 1
    fi
done

echo "警告: 这将覆盖现有数据！"
echo "恢复日期: ${RESTORE_DATE}"
read -p "确认恢复? (yes/no): " confirm

if [ "$confirm" != "yes" ]; then
    echo "恢复取消"
    exit 0
fi

# 停止应用服务（保持数据库和存储服务运行）
echo "停止应用服务..."
docker-compose stop backend frontend

# 1. 恢复PostgreSQL
echo "恢复PostgreSQL数据库..."
docker exec -i law-postgres pg_restore -U law_admin --clean --if-exists --dbname=law_firm < ${BACKUP_DIR}/daily/db_${RESTORE_DATE}.dump

# 2. 恢复MinIO
echo "恢复MinIO数据..."
docker stop minio
docker run --rm \
    -v minio_data:/data \
    -v ${BACKUP_DIR}/daily/minio_${RESTORE_DATE}.tar.gz:/backup.tar.gz \
    alpine sh -c "rm -rf /data/* && tar xzf /backup.tar.gz -C / && chown -R 1001:1001 /data"
docker start minio

# 3. 恢复Redis
echo "恢复Redis数据..."
docker stop redis
docker cp ${BACKUP_DIR}/daily/redis_${RESTORE_DATE}.rdb redis:/data/dump.rdb
docker start redis

# 4. 恢复配置
echo "恢复配置..."
tar xzf ${BACKUP_DIR}/daily/config_${RESTORE_DATE}.tar.gz -C /

# 5. 启动服务
echo "启动服务..."
docker-compose start

echo "恢复完成！"
echo "建议检查数据完整性后重启应用: docker-compose restart"

#!/bin/bash
# deploy.sh - 一键部署脚本

set -e

echo "==============================="
echo "智慧律所系统一键部署脚本"
echo "==============================="

# 检查环境
check_dependencies() {
    echo "检查依赖..."
    
    if ! command -v docker &> /dev/null; then
        echo "安装Docker..."
        curl -fsSL https://get.docker.com | bash
        systemctl enable docker
        systemctl start docker
    fi
    
    if ! command -v docker-compose &> /dev/null; then
        echo "安装Docker Compose..."
        curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
        chmod +x /usr/local/bin/docker-compose
    fi
}

# 配置环境
setup_environment() {
    echo "配置环境..."
    
    # 创建目录结构
    mkdir -p /opt/law-firm/{data,logs,backup,ssl}
    
    # 设置权限
    chmod -R 755 /opt/law-firm
    
    # 设置时区
    timedatectl set-timezone Asia/Shanghai
}

# 部署服务
deploy_services() {
    echo "部署服务..."
    
    cd /opt/law-firm
    
    # 复制配置文件
    cp -r config/* ./
    
    # 启动服务
    docker-compose up -d
    
    # 等待服务启动
    echo "等待服务启动..."
    sleep 30
    
    # 检查服务状态
    docker-compose ps
}

# 初始化系统
initialize_system() {
    echo "初始化系统..."
    
    # 等待数据库就绪
    echo "等待数据库就绪..."
    sleep 10
    
    # 运行数据库初始化脚本
    docker exec law-postgres psql -U law_admin -d law_firm -f /docker-entrypoint-initdb.d/01-create-tables.sql
    docker exec law-postgres psql -U law_admin -d law_firm -f /docker-entrypoint-initdb.d/02-init-data.sql
    
    # 配置备份定时任务
    cp backup/setup-cron.sh /opt/law-firm/
    bash /opt/law-firm/setup-cron.sh
    
    echo "系统初始化完成！"
}

# 显示部署信息
show_deployment_info() {
    echo ""
    echo "==============================="
    echo "部署完成！"
    echo "==============================="
    echo ""
    echo "服务地址:"
    echo "  - 前端管理界面: https://lawfirm.local"
    echo "  - 后端API: https://lawfirm.local/api"
    echo "  - MinIO控制台: http://lawfirm.local:9001"
    echo "  - 数据库管理: postgresql://lawfirm.local:5432"
    echo ""
    echo "默认账号:"
    echo "  - 超级管理员: admin / Admin123456"
    echo ""
    echo "下一步操作:"
    echo "  1. 访问 https://lawfirm.local 登录系统"
    echo "  2. 修改默认密码"
    echo "  3. 配置律所基础信息"
    echo "  4. 添加用户和权限"
    echo ""
    echo "查看服务状态: docker-compose ps"
    echo "查看服务日志: docker-compose logs -f"
    echo ""
}

# 主流程
main() {
    check_dependencies
    setup_environment
    deploy_services
    initialize_system
    show_deployment_info
}

main "$@"

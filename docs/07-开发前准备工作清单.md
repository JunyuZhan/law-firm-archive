### 3.3 Docker环境准备 ⏳

#### 3.3.1 创建docker-compose文件

```yaml
# docker-compose.yml
version: '3.8'

services:
  postgres:
    image: postgres:15-alpine
    container_name: law-postgres
    environment:
      POSTGRES_DB: law_firm
      POSTGRES_USER: law_admin
      POSTGRES_PASSWORD: ${DB_PASSWORD:-Law@2026}
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./database/init:/docker-entrypoint-initdb.d
    networks:
      - law-network

  redis:
    image: redis:7-alpine
    container_name: law-redis
    command: redis-server --requirepass ${REDIS_PASSWORD:-Redis@2026}
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    networks:
      - law-network

  minio:
    image: minio/minio:latest
    container_name: law-minio
    command: server /data --console-address ":9001"
    environment:
      MINIO_ROOT_USER: ${MINIO_USER:-minioadmin}
      MINIO_ROOT_PASSWORD: ${MINIO_PASSWORD:-Minio@2026}
    ports:
      - "9000:9000"
      - "9001:9001"
    volumes:
      - minio_data:/data
    networks:
      - law-network

  elasticsearch:
    image: elasticsearch:8.11.0
    container_name: law-elasticsearch
    environment:
      - discovery.type=single-node
      - xpack.security.enabled=false
      - "ES_JAVA_OPTS=-Xms512m -Xmx512m"
    ports:
      - "9200:9200"
    volumes:
      - es_data:/usr/share/elasticsearch/data
    networks:
      - law-network

volumes:
  postgres_data:
  redis_data:
  minio_data:
  es_data:

networks:
  law-network:
    driver: bridge
```

```bash
#!/bin/bash
# init-project.sh - 项目初始化脚本

set -e

echo "========================================="
echo "智慧律所管理系统 - 项目初始化"
echo "========================================="

# 1. 创建项目目录
PROJECT_ROOT=$(pwd)/law-firm-management
mkdir -p $PROJECT_ROOT
cd $PROJECT_ROOT

# 2. 创建目录结构
mkdir -p backend frontend database/{init,migration,seed} docker docs api

# 3. 初始化Git
git init
echo "# 智慧律所管理系统" > README.md

# 4. 创建.gitignore
cat > .gitignore << 'EOF'
# Java
*.class
*.jar
*.war
target/
.idea/
*.iml

# Node
node_modules/
dist/
.pnpm-store/

# Environment
.env
*.local

# Logs
*.log
logs/

# OS
.DS_Store
Thumbs.db
EOF

# 5. 创建docker-compose.yml
cat > docker/docker-compose.yml << 'EOF'
version: '3.8'
services:
  postgres:
    image: postgres:15-alpine
    container_name: law-postgres
    environment:
      POSTGRES_DB: law_firm
      POSTGRES_USER: law_admin
      POSTGRES_PASSWORD: Law@2026
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
  
  redis:
    image: redis:7-alpine
    container_name: law-redis
    command: redis-server --requirepass Redis@2026
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data

  minio:
    image: minio/minio:latest
    container_name: law-minio
    command: server /data --console-address ":9001"
    environment:
      MINIO_ROOT_USER: minioadmin
      MINIO_ROOT_PASSWORD: Minio@2026
    ports:
      - "9000:9000"
      - "9001:9001"
    volumes:
      - minio_data:/data

volumes:
  postgres_data:
  redis_data:
  minio_data:
EOF

# 6. 创建环境变量模板
cat > .env.example << 'EOF'
# 数据库
DB_HOST=localhost
DB_PORT=5432
DB_NAME=law_firm
DB_USER=law_admin
DB_PASSWORD=Law@2026

# Redis
REDIS_HOST=localhost
REDIS_PORT=6379
REDIS_PASSWORD=Redis@2026

# MinIO
MINIO_ENDPOINT=localhost:9000
MINIO_ACCESS_KEY=minioadmin
MINIO_SECRET_KEY=Minio@2026

# JWT
JWT_SECRET=your-256-bit-secret-key-change-in-production
JWT_EXPIRE_HOURS=24
EOF

cp .env.example .env

echo ""
echo "========================================="
echo "初始化完成！"
echo "========================================="
echo ""
echo "下一步操作："
echo "1. cd $PROJECT_ROOT"
echo "2. cd docker && docker-compose up -d  # 启动基础服务"
echo "3. 创建后端项目：使用 Spring Initializr"
echo "4. 创建前端项目：git clone vben-admin"
echo ""
```

### 9.1 开发前必须完成的核心任务

| 序号 | 任务 | 负责人 | 预计时间 | 优先级 |
|------|------|--------|----------|--------|
| 1 | 创建Git仓库 | 技术负责人 | 0.5天 | 高 |
| 2 | 搭建后端项目骨架 | 后端开发 | 1天 | 高 |
| 3 | 搭建前端项目骨架（Vben Admin） | 前端开发 | 0.5天 | 高 |
| 4 | Docker环境配置 | 后端开发 | 0.5天 | 高 |
| 5 | 数据库SQL脚本整理 | 后端开发 | 1天 | 高 |
| 6 | 开发环境验证 | 全员 | 0.5天 | 高 |
| 7 | 团队启动会 | 项目经理 | 0.5天 | 高 |
| 8 | 第一Sprint计划 | 项目经理 | 0.5天 | 高 |

**总计：约4-5个工作日**

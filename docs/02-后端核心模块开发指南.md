```java
import com.lawfirm.common.result.Result;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpStatus;
import org.springframework.validation.BindException;
import org.springframework.validation.FieldError;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ExceptionHandler;

import org.springframework.web.bind.annotation.RestControllerAdvice;

import javax.validation.ConstraintViolationException;

@Slf4j
@RestControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(BusinessException.class)
    public Result<Void> handleBusinessException(BusinessException e) {
        log.warn("业务异常: {}", e.getMessage());
        return Result.error(e.getCode(), e.getMessage());
    }

    @ExceptionHandler(MethodArgumentNotValidException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public Result<Void> handleMethodArgumentNotValidException(MethodArgumentNotValidException e) {
        String message = e.getBindingResult().getFieldErrors().stream()
                .map(FieldError::getDefaultMessage)
                .collect(Collectors.joining(", "));
        log.warn("参数校验失败: {}", message);
        return Result.error("400", message);
    }

    @ExceptionHandler(BindException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public Result<Void> handleBindException(BindException e) {
        String message = e.getBindingResult().getFieldErrors().stream()
                .map(FieldError::getDefaultMessage)
                .collect(Collectors.joining(", "));
        log.warn("参数绑定失败: {}", message);
        return Result.error("400", message);
    }

    @ExceptionHandler(ConstraintViolationException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public Result<Void> handleConstraintViolationException(ConstraintViolationException e) {
        String message = e.getConstraintViolations().stream()
                .map(violation -> violation.getPropertyPath() + ": " + violation.getMessage())
                .collect(Collectors.joining(", "));
        log.warn("参数验证失败: {}", message);
        return Result.error("400", message);
    }

    @ExceptionHandler(Exception.class)
    @ResponseStatus(HttpStatus.INTERNAL_SERVER_ERROR)
    public Result<Void> handleException(Exception e, HttpServletRequest request) {
        log.error("系统异常: {} {}", request.getMethod(), request.getRequestURI(), e);
        return Result.error("500", "系统异常，请稍后重试");
    }
}
```

```java
// application/system/service/AuthService.java
package com.lawfirm.application.system.service;

import com.lawfirm.common.exception.BusinessException;
import com.lawfirm.domain.system.entity.User;
import com.lawfirm.domain.system.repository.UserRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
@Slf4j
@RequiredArgsConstructor
public class AuthService {
    
    private final UserRepository userRepository;
    private final PasswordEncoder passwordEncoder;
    private final JwtTokenProvider jwtTokenProvider;
    
    @Transactional
    public LoginResult login(LoginCommand command) {
        // 1. 验证用户是否存在
        User user = userRepository.findByUsername(command.getUsername())
                .orElseThrow(() -> new BusinessException("用户名或密码错误"));
        
        // 2. 验证用户状态
        if (!user.isActive()) {
            throw new BusinessException("用户已被禁用，请联系管理员");
        }
        
        // 3. 验证密码
        if (!passwordEncoder.matches(command.getPassword(), user.getPassword())) {
            // 记录登录失败次数
            handleFailedLogin(user);
            throw new BusinessException("用户名或密码错误");
        }
        
        // 4. 重置登录失败次数
        resetFailedLogin(user);
        
        // 5. 更新最后登录时间
        userRepository.updateLastLogin(user.getId());
        
        // 6. 生成Token
        String token = jwtTokenProvider.generateToken(user);
        
        // 7. 获取用户权限
        List<String> permissions = permissionService.getUserPermissions(user.getId());
        
        return LoginResult.builder()
                .token(token)
                .userId(user.getId())
                .username(user.getUsername())
                .realName(user.getRealName())
                .avatar(user.getAvatarUrl())
                .permissions(permissions)
                .build();
    }
    
    private void handleFailedLogin(User user) {
        // 实现登录失败逻辑
        log.warn("用户 {} 登录失败", user.getUsername());
    }
    
    private void resetFailedLogin(User user) {
        // 重置登录失败计数
    }
}
```

```java
// infrastructure/security/JwtTokenProvider.java
package com.lawfirm.infrastructure.security;

import com.lawfirm.domain.system.entity.User;
import io.jsonwebtoken.*;
import io.jsonwebtoken.security.Keys;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.security.authentication.UsernamePasswordAuthenticationToken;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.authority.SimpleGrantedAuthority;
import org.springframework.stereotype.Component;

import javax.crypto.SecretKey;
import java.util.Date;
import java.util.List;
import java.util.stream.Collectors;

@Slf4j
@Component
public class JwtTokenProvider {
    
    @Value("${jwt.secret}")
    private String jwtSecret;
    
    @Value("${jwt.expiration}")
    private long jwtExpiration;
    
    private SecretKey getSigningKey() {
        return Keys.hmacShaKeyFor(jwtSecret.getBytes());
    }
    
    public String generateToken(User user) {
        Date now = new Date();
        Date expiryDate = new Date(now.getTime() + jwtExpiration);
        
        return Jwts.builder()
                .setSubject(user.getId().toString())
                .claim("username", user.getUsername())
                .claim("realName", user.getRealName())
                .setIssuedAt(now)
                .setExpiration(expiryDate)
                .signWith(getSigningKey(), SignatureAlgorithm.HS512)
                .compact();
    }
    
    public Long getUserIdFromToken(String token) {
        Claims claims = Jwts.parserBuilder()
                .setSigningKey(getSigningKey())
                .build()
                .parseClaimsJws(token)
                .getBody();
        
        return Long.parseLong(claims.getSubject());
    }
    
    public boolean validateToken(String token) {
        try {
            Jwts.parserBuilder()
                .setSigningKey(getSigningKey())
                .build()
                .parseClaimsJws(token);
            return true;
        } catch (SecurityException ex) {
            log.error("Invalid JWT signature");
        } catch (MalformedJwtException ex) {
            log.error("Invalid JWT token");
        } catch (ExpiredJwtException ex) {
            log.error("Expired JWT token");
        } catch (UnsupportedJwtException ex) {
            log.error("Unsupported JWT token");
        } catch (IllegalArgumentException ex) {
            log.error("JWT claims string is empty");
        }
        return false;
    }
    
    public Authentication getAuthentication(String token) {
        Claims claims = Jwts.parserBuilder()
                .setSigningKey(getSigningKey())
                .build()
                .parseClaimsJws(token)
                .getBody();
        
        String userId = claims.getSubject();
        String username = claims.get("username", String.class);
        
        // 从数据库或缓存获取用户权限
        List<SimpleGrantedAuthority> authorities = getAuthorities(userId);
        
        return new UsernamePasswordAuthenticationToken(
                username, null, authorities);
    }
    
    private List<SimpleGrantedAuthority> getAuthorities(String userId) {
        // 实现获取用户权限的逻辑
        return List.of();
    }
}
```

```java
// infrastructure/security/PermissionInterceptor.java
package com.lawfirm.infrastructure.security;

import com.lawfirm.common.annotation.RequirePermission;
import com.lawfirm.common.exception.BusinessException;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.web.method.HandlerMethod;
import org.springframework.web.servlet.HandlerInterceptor;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.lang.reflect.Method;

@Slf4j
@Component
@RequiredArgsConstructor
public class PermissionInterceptor implements HandlerInterceptor {
    
    private final PermissionService permissionService;
    
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {
        if (!(handler instanceof HandlerMethod)) {
            return true;
        }
        
        HandlerMethod handlerMethod = (HandlerMethod) handler;
        Method method = handlerMethod.getMethod();
        
        // 检查是否需要权限
        RequirePermission requirePermission = method.getAnnotation(RequirePermission.class);
        if (requirePermission == null) {
            requirePermission = method.getDeclaringClass().getAnnotation(RequirePermission.class);
        }
        
        if (requirePermission == null) {
            return true;
        }
        
        // 获取当前用户
        Long userId = SecurityUtils.getCurrentUserId();
        if (userId == null) {
            throw new BusinessException("用户未登录");
        }
        
        // 检查权限
        boolean hasPermission = permissionService.hasPermission(userId, requirePermission.value());
        if (!hasPermission) {
            throw new BusinessException("没有操作权限");
        }
        
        return true;
    }
}
```

```java
package com.lawfirm.application.system.service;

import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.lawfirm.common.exception.BusinessException;
import com.lawfirm.common.result.PageResult;
import com.lawfirm.domain.system.entity.User;
import com.lawfirm.domain.system.repository.UserRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
@RequiredArgsConstructor
public class UserAppService {
    
    private final UserRepository userRepository;
    private final PasswordEncoder passwordEncoder;
    
    @Transactional
    public UserDTO create(CreateUserCommand command) {
        // 验证用户名唯一性
        if (userRepository.existsByUsername(command.getUsername())) {
            throw new BusinessException("用户名已存在");
        }
        
        // 创建用户实体
        User user = User.builder()
                .username(command.getUsername())
                .password(passwordEncoder.encode(command.getPassword()))
                .realName(command.getRealName())
                .email(command.getEmail())
                .phone(command.getPhone())
                .departmentId(command.getDepartmentId())
                .position(command.getPosition())
                .employeeNo(command.getEmployeeNo())
                .lawyerLicenseNo(command.getLawyerLicenseNo())
                .joinDate(command.getJoinDate())
                .status("ACTIVE")
                .build();
        
        // 保存用户
        userRepository.save(user);
        
        // 分配角色
        if (command.getRoleIds() != null && !command.getRoleIds().isEmpty()) {
            userRoleService.assignRoles(user.getId(), command.getRoleIds());
        }
        
        return UserAssembler.toDTO(user);
    }
    
    @Transactional
    public UserDTO update(Long id, UpdateUserCommand command) {
        User user = userRepository.findById(id)
                .orElseThrow(() -> new BusinessException("用户不存在"));
        
        // 更新基本信息
        user.setRealName(command.getRealName());
        user.setEmail(command.getEmail());
        user.setPhone(command.getPhone());
        user.setDepartmentId(command.getDepartmentId());
        user.setPosition(command.getPosition());
        user.setLawyerLicenseNo(command.getLawyerLicenseNo());
        
        // 更新密码（如果提供了新密码）
        if (StringUtils.isNotBlank(command.getNewPassword())) {
            user.setPassword(passwordEncoder.encode(command.getNewPassword()));
        }
        
        userRepository.updateById(user);
        
        // 更新角色
        if (command.getRoleIds() != null) {
            userRoleService.updateUserRoles(id, command.getRoleIds());
        }
        
        return UserAssembler.toDTO(user);
    }
    
    public PageResult<UserDTO> queryPage(UserQueryRequest request) {
        Page<User> page = userRepository.queryPage(
                request.getPage(), 
                request.getPageSize(),
                request.getKeyword(),
                request.getDepartmentId(),
                request.getStatus()
        );
        
        List<UserDTO> items = page.getRecords().stream()
                .map(UserAssembler::toDTO)
                .collect(Collectors.toList());
        
        return PageResult.of(page.getTotal(), items, request.getPage(), request.getPageSize());
    }
}
```

```java
// application/system/service/PermissionService.java
package com.lawfirm.application.system.service;

import com.lawfirm.common.exception.BusinessException;
import com.lawfirm.domain.system.entity.Permission;
import com.lawfirm.domain.system.entity.Role;
import com.lawfirm.domain.system.repository.PermissionRepository;
import com.lawfirm.domain.system.repository.RolePermissionRepository;
import com.lawfirm.domain.system.repository.RoleRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.cache.annotation.Cacheable;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.stream.Collectors;

@Service
@RequiredArgsConstructor
public class PermissionService {
    
    private final PermissionRepository permissionRepository;
    private final RoleRepository roleRepository;
    private final RolePermissionRepository rolePermissionRepository;
    
    @Cacheable(value = "userPermissions", key = "#userId")
    public List<String> getUserPermissions(Long userId) {
        // 获取用户角色
        List<Long> roleIds = userRoleRepository.findRoleIdsByUserId(userId);
        
        // 获取角色权限
        return rolePermissionRepository.findPermissionCodesByRoleIds(roleIds);
    }
    
    public boolean hasPermission(Long userId, String permissionCode) {
        List<String> permissions = getUserPermissions(userId);
        return permissions.contains(permissionCode);
    }
    
    @Transactional
    public RoleDTO createRole(CreateRoleCommand command) {
        // 验证角色代码唯一性
        if (roleRepository.existsByRoleCode(command.getRoleCode())) {
            throw new BusinessException("角色代码已存在");
        }
        
        Role role = Role.builder()
                .roleCode(command.getRoleCode())
                .roleName(command.getRoleName())
                .description(command.getDescription())
                .dataScope(command.getDataScope())
                .sortOrder(command.getSortOrder())
                .status("ACTIVE")
                .build();
        
        roleRepository.save(role);
        
        // 分配权限
        if (command.getPermissionIds() != null && !command.getPermissionIds().isEmpty()) {
            assignPermissionsToRole(role.getId(), command.getPermissionIds());
        }
        
        return RoleAssembler.toDTO(role);
    }
    
    @Transactional
    public void assignPermissionsToRole(Long roleId, List<Long> permissionIds) {
        // 验证角色存在
        Role role = roleRepository.findById(roleId)
                .orElseThrow(() -> new BusinessException("角色不存在"));
        
        // 删除原有权限
        rolePermissionRepository.deleteByRoleId(roleId);
        
        // 添加新权限
        List<RolePermission> rolePermissions = permissionIds.stream()
                .map(permissionId -> RolePermission.builder()
                        .roleId(roleId)
                        .permissionId(permissionId)
                        .build())
                .collect(Collectors.toList());
        
        rolePermissionRepository.saveBatch(rolePermissions);
    }
}
```

```java
// infrastructure/security/DataScopeHandler.java
@Component
public class DataScopeHandler {
    
    @Autowired
    private UserService userService;
    
    /**
     * 根据数据范围生成SQL过滤条件
     */
    public String buildDataScopeFilter(Long userId, String tableAlias, DataScope dataScope) {
        if (dataScope == null) {
            // 获取用户默认数据范围
            dataScope = userService.getUserDataScope(userId);
        }
        
        String alias = StringUtils.isNotEmpty(tableAlias) ? tableAlias + "." : "";
        
        switch (dataScope) {
            case ALL:
                return "";  // 无过滤条件
                
            case DEPT:
                Long deptId = userService.getUserDeptId(userId);
                return String.format(" AND %sdept_id = %d", alias, deptId);
                
            case TEAM:
                Long teamId = userService.getUserTeamId(userId);
                return String.format(" AND %steam_id = %d", alias, teamId);
                
            case SELF:
                return String.format(" AND %screated_by = %d", alias, userId);
                
            case PARTICIPATED:
                return String.format(
                    " AND %smatter_id IN (SELECT matter_id FROM matter_team WHERE user_id = %d)",
                    alias, userId
                );
                
            case ASSIGNED:
                return String.format(" AND %sassigned_to = %d", alias, userId);
                
            default:
                return String.format(" AND %screated_by = %d", alias, userId);
        }
    }
}
```

### 4.2 利益冲突审查领域服务

```java
// domain/client/service/ConflictCheckDomainService.java
package com.lawfirm.domain.client.service;

import com.lawfirm.common.exception.BusinessException;
import com.lawfirm.domain.client.entity.ConflictCheck;
import com.lawfirm.domain.client.repository.ClientRepository;
import com.lawfirm.domain.client.repository.ConflictCheckRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.time.LocalDate;
import java.util.List;

@Slf4j
@Service
@RequiredArgsConstructor
public class ConflictCheckDomainService {
    
    private final ConflictCheckRepository conflictCheckRepository;
    private final ClientRepository clientRepository;
    
    /**
     * 执行利益冲突审查
     */
    public ConflictCheckResult checkConflict(ConflictCheckCommand command) {
        // 1. 验证客户存在
        if (!clientRepository.existsById(command.getClientId())) {
            throw new BusinessException("客户不存在");
        }
        
        // 2. 自动检索冲突
        List<ConflictRecord> autoConflicts = autoCheckConflict(
                command.getClientName(),
                command.getOpposingParty(),
                command.getRelatedPersons()
        );
        
        // 3. 创建审查记录
        ConflictCheck conflictCheck = ConflictCheck.builder()
                .checkNo(generateCheckNo())
                .applicantId(command.getApplicantId())
                .clientName(command.getClientName())
                .clientType(command.getClientType())
                .opposingPartyName(command.getOpposingParty())
                .relatedPersons(command.getRelatedPersons())
                .businessType(command.getBusinessType())
                .businessDescription(command.getBusinessDescription())
                .checkMethod(autoConflicts.isEmpty() ? "AUTO" : "MANUAL")
                .checkResult(autoConflicts.isEmpty() ? "PASS" : "CONFLICT")
                .conflictDetails(autoConflicts.isEmpty() ? null : formatConflictDetails(autoConflicts))
                .riskLevel(autoConflicts.isEmpty() ? "LOW" : "HIGH")
                .validUntil(LocalDate.now().plusDays(30))
                .build();
        
        conflictCheckRepository.save(conflictCheck);
        
        return ConflictCheckResult.builder()
                .checkId(conflictCheck.getId())
                .checkNo(conflictCheck.getCheckNo())
                .result(conflictCheck.getCheckResult())
                .conflicts(autoConflicts)
                .riskLevel(conflictCheck.getRiskLevel())
                .build();
    }
    
    /**
     * 自动冲突检查
     */
    private List<ConflictRecord> autoCheckConflict(String clientName, 
                                                  String opposingParty, 
                                                  List<String> relatedPersons) {
        List<ConflictRecord> conflicts = new ArrayList<>();
        
        // 检查相对方冲突
        if (opposingParty != null && !opposingParty.trim().isEmpty()) {
            List<ConflictCheck> existingChecks = conflictCheckRepository
                    .findByClientNameOrOpposingParty(opposingParty, clientName);
            
            if (!existingChecks.isEmpty()) {
                conflicts.add(ConflictRecord.builder()
                        .type("OPPOSING_PARTY_CONFLICT")
                        .description("相对方 " + opposingParty + " 与已有客户存在关联")
                        .relatedChecks(existingChecks)
                        .build());
            }
        }
        
        // 检查关联人冲突
        if (relatedPersons != null && !relatedPersons.isEmpty()) {
            for (String person : relatedPersons) {
                List<ConflictCheck> personConflicts = conflictCheckRepository
                        .findByRelatedPerson(person);
                
                if (!personConflicts.isEmpty()) {
                    conflicts.add(ConflictRecord.builder()
                            .type("RELATED_PERSON_CONFLICT")
                            .description("关联人 " + person + " 与已有客户存在关联")
                            .relatedChecks(personConflicts)
                            .build());
                }
            }
        }
        
        return conflicts;
    }
    
    /**
     * 验证利冲审查通过
     */
    public void ensurePassed(Long clientId, String opposingParty) {
        List<ConflictCheck> validChecks = conflictCheckRepository
                .findValidPassedChecks(clientId, opposingParty);
        
        if (validChecks.isEmpty()) {
            throw new BusinessException("未找到有效的利益冲突审查记录，请先进行利冲审查");
        }
    }
    
    private String generateCheckNo() {
        return "CC" + System.currentTimeMillis();
    }
}
```

```java
            (client.getLegalPerson() == null || client.getLegalPerson().trim().isEmpty())) {
            throw new BusinessException("企业客户必须填写法定代表人");
        }
        
        // 4. 保存更新
        clientRepository.updateById(client);
        
        // 5. 更新联系人
        updateContacts(id, command.getContacts());
        
        return ClientAssembler.toDTO(client);
    }
    
    public PageResult<ClientDTO> queryPage(ClientQueryRequest request) {
        Page<Client> page = clientRepository.queryPage(
                request.getPage(),
                request.getPageSize(),
                request.getKeyword(),
                request.getClientType(),
                request.getImportanceLevel(),
                request.getSourceChannel(),
                request.getCooperationStatus()
        );
        
        List<ClientDTO> items = page.getRecords().stream()
                .map(client -> {
                    ClientDTO dto = ClientAssembler.toDTO(client);
                    // 加载联系人
                    List<Contact> contacts = contactRepository.findByClientId(client.getId());
                    dto.setContacts(ContactAssembler.toDTOList(contacts));
                    return dto;
                })
                .collect(Collectors.toList());
        
        return PageResult.of(page.getTotal(), items, request.getPage(), request.getPageSize());
    }
```

```java
        // 6. 添加参与人

                )
        );

        return dto;
    }
```

```java
// application/finance/service/PaymentReconciliationService.java
package com.lawfirm.application.finance.service;

import com.lawfirm.common.exception.BusinessException;
import com.lawfirm.domain.finance.entity.Payment;
import com.lawfirm.infrastructure.external.ocr.OcrService;
import com.lawfirm.domain.system.repository.PaymentRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import org.springframework.web.multipart.MultipartFile;

import java.math.BigDecimal;
import java.time.LocalDate;
import java.util.Comparator;
import java.util.List;
import java.util.stream.Collectors;

@Slf4j
@Service
@RequiredArgsConstructor
public class PaymentReconciliationService {
    
    private final PaymentRepository paymentRepository;
    private final OcrService ocrService;
    private final MatterRepository matterRepository;
    
    /**
     * 上传银行回单并智能匹配
     */
    @Transactional
    public ReconciliationResult uploadAndReconcile(MultipartFile file, Long uploaderId) {
        // 1. OCR识别
        OcrResult ocrResult = ocrService.recognizeBankStatement(file);
        
        // 2. 智能匹配项目
        MatchResult matchResult = intelligentMatch(ocrResult);
        
        // 3. 创建收款记录
        Payment payment = createPaymentRecord(ocrResult, uploaderId, matchResult);
        
        // 4. 如果匹配度足够高，自动核销
        if (matchResult.getRecommended() != null && 
            matchResult.getRecommended().getScore() > 0.7) {
            autoReconcile(payment, matchResult.getRecommended());
        }
        
        return ReconciliationResult.builder()
                .paymentId(payment.getId())
                .ocrResult(ocrResult)
                .matchResult(matchResult)
                .build();
    }
    
    /**
     * 智能匹配项目
     */
    private MatchResult intelligentMatch(OcrResult ocrResult) {
        List<MatchCandidate> candidates = new ArrayList<>();
        
        // 1. 按金额匹配
        if (ocrResult.getAmount() != null) {
            List<Matter> amountMatches = matterRepository.findByPendingAmount(
                    ocrResult.getAmount().multiply(new BigDecimal("0.9")), // 90%
                    ocrResult.getAmount().multiply(new BigDecimal("1.1"))  // 110%
            );
            
            for (Matter matter : amountMatches) {
                double score = calculateMatchScore(ocrResult, matter);
                candidates.add(MatchCandidate.builder()
                        .matter(matter)
                        .score(score)
                        .reasons(List.of("金额匹配"))
                        .build());
            }
        }
        
        // 2. 按付款方名称匹配
        if (ocrResult.getPayer() != null) {
            List<Matter> payerMatches = matterRepository.findByClientNameLike(ocrResult.getPayer());
            
            for (Matter matter : payerMatches) {
                double score = calculateMatchScore(ocrResult, matter);
                candidates.add(MatchCandidate.builder()
                        .matter(matter)
                        .score(score)
                        .reasons(List.of("付款方名称匹配"))
                        .build());
            }
        }
        
        // 排序并返回
        candidates.sort(Comparator.comparing(MatchCandidate::getScore).reversed());
        
        MatchCandidate recommended = null;
        if (!candidates.isEmpty() && candidates.get(0).getScore() > 0.5) {
            recommended = candidates.get(0);
        }
        
        return MatchResult.builder()
                .recommended(recommended)
                .candidates(candidates)
                .build();
    }
    
    /**
     * 计算匹配分数
     */
    private double calculateMatchScore(OcrResult ocrResult, Matter matter) {
        double score = 0.0;
        
        // 金额匹配 (权重0.6)
        if (ocrResult.getAmount() != null && matter.getPendingAmount() != null) {
            BigDecimal diff = ocrResult.getAmount().subtract(matter.getPendingAmount()).abs();
            BigDecimal ratio = diff.divide(matter.getPendingAmount(), 4, BigDecimal.ROUND_HALF_UP);
            if (ratio.compareTo(new BigDecimal("0.1")) <= 0) { // 10%以内
                score += 0.6;
            } else if (ratio.compareTo(new BigDecimal("0.2")) <= 0) { // 20%以内
                score += 0.3;
            }
        }
        
        // 付款方匹配 (权重0.4)
        if (ocrResult.getPayer() != null && matter.getClientName() != null) {
            if (matter.getClientName().contains(ocrResult.getPayer()) || 
                ocrResult.getPayer().contains(matter.getClientName())) {
                score += 0.4;
            }
        }
        
        return score;
    }
    
    /**
     * 创建收款记录
     */
    private Payment createPaymentRecord(OcrResult ocrResult, Long uploaderId, MatchResult matchResult) {
        Payment payment = Payment.builder()
                .paymentNo(generatePaymentNo())
                .amount(ocrResult.getAmount())
                .currency("CNY")
                .paymentDate(ocrResult.getDate() != null ? ocrResult.getDate() : LocalDate.now())
                .payerName(ocrResult.getPayer())
                .bankStatementUrl(ocrResult.getFileUrl())
                .ocrData(ocrResult.getRawData())
                .matchScore(matchResult.getRecommended() != null ? 
                           matchResult.getRecommended().getScore() : 0.0)
                .matchMethod(matchResult.getRecommended() != null ? "AUTO" : "MANUAL")
                .status("PENDING")
                .createdBy(uploaderId)
                .build();
        
        // 如果自动匹配成功，关联项目
        if (matchResult.getRecommended() != null) {
            payment.setMatterId(matchResult.getRecommended().getMatter().getId());
        }
        
        paymentRepository.save(payment);
        return payment;
    }
    
    /**
     * 自动核销
     */
    private void autoReconcile(Payment payment, MatchCandidate match) {
        payment.setStatus("CONFIRMED");
        payment.setConfirmedBy(1L); // 系统自动确认
        payment.setConfirmedAt(LocalDateTime.now());
        
        paymentRepository.updateById(payment);
        
        // 触发提成计算
        commissionService.calculateCommission(payment.getId());
        
        log.info("自动核销成功: paymentId={}, matterId={}, amount={}", 
                payment.getId(), match.getMatter().getId(), payment.getAmount());
    }
```

```java
// domain/finance/service/CommissionCalculator.java
package com.lawfirm.domain.finance.service;

import com.lawfirm.common.exception.BusinessException;
import com.lawfirm.domain.finance.entity.*;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.math.BigDecimal;
import java.math.RoundingMode;
import java.util.ArrayList;
import java.util.List;

@Slf4j
@Service
public class CommissionCalculator {
    
    /**
     * 计算提成
     */
    public CommissionResult calculate(Payment payment, Matter matter, 
                                       List<MatterParticipant> participants,
                                       CommissionRule rule) {
        
        // 1. 验证输入
        validateCalculationInput(payment, matter, participants, rule);
        
        // 2. 计算净收入
        CommissionCalculation calculation = calculateNetAmount(payment, rule);
        
        // 3. 计算每个参与人的提成
        List<CommissionDetail> details = new ArrayList<>();
        
        for (MatterParticipant participant : participants) {
            CommissionDetail detail = calculateParticipantCommission(
                    participant, calculation, rule);
            details.add(detail);
        }
        
        // 4. 验证提成总和
        validateCommissionTotal(details, calculation.getNetAmount());
        
        return CommissionResult.builder()
                .paymentId(payment.getId())
                .matterId(matter.getId())
                .calculation(calculation)
                .details(details)
                .build();
    }
    
    /**
     * 计算净收入
     */
    private CommissionCalculation calculateNetAmount(Payment payment, CommissionRule rule) {
        BigDecimal grossAmount = payment.getAmount();
        
        // 税费
        BigDecimal taxAmount = grossAmount.multiply(rule.getTaxRate())
                .setScale(2, RoundingMode.HALF_UP);
        
        // 管理费
        BigDecimal managementFee = grossAmount.multiply(rule.getManagementFeeRate())
                .setScale(2, RoundingMode.HALF_UP);
        
        // 办案成本（从payment中获取或默认为0）
        BigDecimal costAmount = payment.getCostAmount() != null ? 
                payment.getCostAmount() : BigDecimal.ZERO;
        
        // 净收入 = 毛收入 - 税费 - 管理费 - 办案成本
        BigDecimal netAmount = grossAmount.subtract(taxAmount)
                .subtract(managementFee)
                .subtract(costAmount);
        
        return CommissionCalculation.builder()
                .grossAmount(grossAmount)
                .taxAmount(taxAmount)
                .managementFeeAmount(managementFee)
                .costAmount(costAmount)
                .netAmount(netAmount)
                .build();
    }
    
    /**
     * 计算参与人提成
     */
    private CommissionDetail calculateParticipantCommission(MatterParticipant participant,
                                                           CommissionCalculation calculation,
                                                           CommissionRule rule) {
        // 分配比例
        BigDecimal distributionRatio = participant.getDistributionRatio();
        
        // 该参与人的净收入份额
        BigDecimal participantNetAmount = calculation.getNetAmount()
                .multiply(distributionRatio)
                .setScale(2, RoundingMode.HALF_UP);
        
        // 根据阶梯计算提成比例
        BigDecimal commissionRate = calculateTieredRate(participantNetAmount, rule.getRateTiers());
        
        // 提成金额
        BigDecimal commissionAmount = participantNetAmount.multiply(commissionRate)
                .setScale(2, RoundingMode.HALF_UP);
        
        return CommissionDetail.builder()
                .lawyerId(participant.getUserId())
                .grossAmount(calculation.getGrossAmount().multiply(distributionRatio)
                        .setScale(2, RoundingMode.HALF_UP))
                .taxAmount(calculation.getTaxAmount().multiply(distributionRatio)
                        .setScale(2, RoundingMode.HALF_UP))
                .managementFeeAmount(calculation.getManagementFeeAmount().multiply(distributionRatio)
                        .setScale(2, RoundingMode.HALF_UP))
                .costAmount(calculation.getCostAmount().multiply(distributionRatio)
                        .setScale(2, RoundingMode.HALF_UP))
                .netAmount(participantNetAmount)
                .distributionRatio(distributionRatio)
                .commissionRate(commissionRate)
                .commissionAmount(commissionAmount)
                .build();
    }
    
    /**
     * 阶梯提成比例计算
     */
    private BigDecimal calculateTieredRate(BigDecimal amount, List<RateTier> tiers) {
        for (RateTier tier : tiers) {
            if (amount.compareTo(tier.getMinAmount()) >= 0 && 
                (tier.getMaxAmount() == null || amount.compareTo(tier.getMaxAmount()) < 0)) {
                return tier.getRate();
            }
        }
        // 默认最高档
        return tiers.get(tiers.size() - 1).getRate();
    }
    
    /**
     * 验证计算输入
     */
    private void validateCalculationInput(Payment payment, Matter matter,
                                         List<MatterParticipant> participants,
                                         CommissionRule rule) {
        if (payment == null || payment.getAmount() == null) {
            throw new BusinessException("收款记录无效");
        }
        
        if (matter == null) {
            throw new BusinessException("项目信息无效");
        }
        
        if (participants == null || participants.isEmpty()) {
            throw new BusinessException("项目参与人信息无效");
        }
        
        if (rule == null || rule.getRateTiers() == null || rule.getRateTiers().isEmpty()) {
            throw new BusinessException("提成规则配置无效");
        }
        
        // 验证分配比例总和为1
        BigDecimal totalRatio = participants.stream()
                .map(MatterParticipant::getDistributionRatio)
                .reduce(BigDecimal.ZERO, BigDecimal::add);
        
        if (totalRatio.compareTo(BigDecimal.ONE) != 0) {
            throw new BusinessException("参与人分配比例总和必须为1");
        }
    }
    
    /**
     * 验证提成总和
     */
    private void validateCommissionTotal(List<CommissionDetail> details, BigDecimal totalNetAmount) {
        BigDecimal totalCommission = details.stream()
                .map(CommissionDetail::getCommissionAmount)
                .reduce(BigDecimal.ZERO, BigDecimal::add);
        
        BigDecimal maxCommission = totalNetAmount.multiply(new BigDecimal("0.5")); // 最多50%
        
        if (totalCommission.compareTo(maxCommission) > 0) {
            throw new BusinessException("提成总额超过净收入的50%，请检查提成规则");
        }
    }
}
```

```java
     * 收集案件相关的所有变量值
     */
    public Map<String, Object> collectVariables(Long matterId) {
        Map<String, Object> variables = new HashMap<>();
        
        // 获取案件信息
        Matter matter = matterRepository.findById(matterId).orElseThrow();
        variables.put("matter.name", matter.getMatterName());
        variables.put("matter.no", matter.getMatterNo());
        
        // 获取客户信息
        Client client = clientRepository.findById(matter.getClientId()).orElseThrow();
        variables.put("client.name", client.getClientName());
        variables.put("client.legalPerson", client.getLegalPerson());
        variables.put("client.address", client.getBusinessAddress());
        
        // 获取律师信息
        User lawyer = userRepository.findById(matter.getMainLawyerId()).orElseThrow();
        variables.put("lawyer.name", lawyer.getRealName());
        variables.put("lawyer.licenseNo", lawyer.getLawyerLicenseNo());
        
        // 获取诉讼扩展信息
        MatterLitigationExt ext = litigationExtRepository.findByMatterId(matterId).orElse(null);
        if (ext != null) {
            variables.put("matter.caseNo", ext.getCaseNo());
            variables.put("matter.caseCause", ext.getCaseCause());
            variables.put("court.name", ext.getCourtName());
        }
        
        // 日期变量
        variables.put("date.today", LocalDate.now().format(DateTimeFormatter.ofPattern("yyyy年MM月dd日")));
        variables.put("date.year", String.valueOf(LocalDate.now().getYear()));
        
        // 律所信息（从系统配置获取）
        variables.put("firm.name", configService.get("firm.name"));
        variables.put("firm.address", configService.get("firm.address"));
        
        return variables;
    }

     * 解析模板变量，生成文档
     */
    public byte[] generateDocument(Long templateId, Long matterId, Map<String, Object> extraVariables) {
        // 1. 获取模板
        DocTemplate template = templateRepository.findById(templateId).orElseThrow();
        
        // 2. 收集变量值
        Map<String, Object> variables = collectVariables(matterId);
        variables.putAll(extraVariables);  // 合并额外变量
        
        // 3. 读取模板文件
        byte[] templateBytes = minioService.getFile(template.getFileUrl());
        
        // 4. 替换变量（支持Word和PDF）

            result = processWordTemplate(templateBytes, variables);
        } else if (template.getFileFormat().equals("pdf")) {
            result = processPdfTemplate(templateBytes, variables);
        } else {
            throw new BusinessException("不支持的模板格式");
        }
        
        return result;
    }
```

```java
        boolean allSameMatter = evidences.stream()
                .allMatch(e -> e.getMatterId().equals(matterId));
        if (!allSameMatter) {
            throw new BusinessException("选中的证据必须属于同一案件");
        }
        
        // 生成清单编号
        String listNo = "EL" + LocalDate.now().format(DateTimeFormatter.ofPattern("yyyyMMdd")) +

                    .matterId(matterId)
                    .listName(command.getListName())

                    .submitDate(command.getSubmitDate())

                evidence.setUpdatedBy(operatorId);
                evidenceRepository.updateById(evidence);
            }
        }
        log.info("调整证据顺序: matterId={}", matterId);
    }
```

证据管理模块支持多种文件类型的预览和缩略图显示，包括图片、PDF、Word、Excel、视频、音频等。

**实现要点**：

1. **文件类型识别**
   - 在`EvidenceAppService.toDTO()`方法中自动识别文件类型
   - 支持的文件类型：
     - `image`: jpg, jpeg, png, gif, bmp, webp
     - `pdf`: pdf
     - `word`: doc, docx
     - `excel`: xls, xlsx
     - `video`: mp4, avi, mov
     - `audio`: mp3, wav, m4a
     - `other`: 其他类型

2. **缩略图生成**
   - 图片文件：直接使用原文件URL作为缩略图（前端可设置固定尺寸显示）
   - 非图片文件：返回null，前端显示对应的文件类型图标

3. **DTO字段扩展**
   ```java
   // EvidenceDTO.java
   private String fileType;        // 文件类型
   private String thumbnailUrl;     // 缩略图URL（仅图片文件）
   ```

4. **后端接口**
   - `GET /api/evidence/{id}/preview` - 获取文件预览信息
   - `GET /api/evidence/{id}/thumbnail` - 获取缩略图URL

5. **前端实现**
   - 证据列表页面显示缩略图列
   - 图片文件：使用`<Image>`组件显示缩略图，支持点击预览
   - 非图片文件：显示文件类型图标（emoji或图标字体）
   - 文件列：显示文件名和文件大小，支持点击预览或下载

6. **预览功能**
   - 图片：使用Ant Design Vue的`Image`组件，支持预览模式
   - PDF：在新窗口打开PDF文件
   - 其他文件：提示下载后查看

**代码示例**：

```java
// EvidenceAppService.java - toDTO方法
private EvidenceDTO toDTO(Evidence evidence) {
    EvidenceDTO dto = new EvidenceDTO();
    // ... 其他字段设置
    
    // 设置文件类型和缩略图
    if (evidence.getFileName() != null) {
        dto.setFileType(getFileType(evidence.getFileName()));
        if (isImageFile(evidence.getFileName()) && evidence.getFileUrl() != null) {
            dto.setThumbnailUrl(evidence.getFileUrl()); // 图片文件使用原文件URL作为缩略图
        }
    }
    
    return dto;

// 文件类型识别
private String getFileType(String fileName) {
    if (fileName == null) return "unknown";
    String lowerName = fileName.toLowerCase();
    if (lowerName.endsWith(".jpg") || lowerName.endsWith(".jpeg") 
        || lowerName.endsWith(".png") || lowerName.endsWith(".gif") 
        || lowerName.endsWith(".bmp") || lowerName.endsWith(".webp")) {
        return "image";
    } else if (lowerName.endsWith(".pdf")) {
        return "pdf";
    } else if (lowerName.endsWith(".doc") || lowerName.endsWith(".docx")) {
        return "word";
    } else if (lowerName.endsWith(".xls") || lowerName.endsWith(".xlsx")) {
        return "excel";
    } else if (lowerName.endsWith(".mp4") || lowerName.endsWith(".avi") 
               || lowerName.endsWith(".mov")) {
        return "video";
    } else if (lowerName.endsWith(".mp3") || lowerName.endsWith(".wav") 
               || lowerName.endsWith(".m4a")) {
        return "audio";
    } else {
        return "other";
    }
}
```

```java
        // 入库

        archiveRepository.updateById(archive);
        
        // 更新库位容量
        location.incrementUsedCapacity();
        locationRepository.updateById(location);
        
        // 记录操作日志
        logOperation(archiveId, "STORE", "档案入库，库位：" + location.getLocationCode(), command.getOperatorId());

            return "15_YEARS";  // 一般诉讼15年
        }
        
        if ("NON_LITIGATION".equals(archiveType)) {
            return "10_YEARS";  // 非诉项目10年
```

```java
package com.lawfirm.application.document.service;

import com.lawfirm.common.exception.BusinessException;
import com.lawfirm.domain.document.entity.Seal;
import com.lawfirm.domain.document.entity.SealApplication;

import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDate;
import java.util.List;

@Service
@RequiredArgsConstructor
public class SealAppService {
    
    private final SealRepository sealRepository;
    private final SealApplicationRepository sealApplicationRepository;
    private final WorkflowService workflowService;
    
    /**
     * 申请用印
     */
    @Transactional
    public SealApplicationDTO apply(SealApplyCommand command) {
        // 1. 验证印章存在且可用
        Seal seal = sealRepository.findById(command.getSealId())
                .orElseThrow(() -> new BusinessException("印章不存在"));
        
        if (!"AVAILABLE".equals(seal.getStatus())) {
            throw new BusinessException("印章当前不可用");
        }
        
        // 2. 验证申请日期
        if (command.getUsageDate().isBefore(LocalDate.now())) {
            throw new BusinessException("用印日期不能早于今天");
        }
        
        // 3. 创建用印申请
        SealApplication application = SealApplication.builder()
                .applicationNo(generateApplicationNo())
                .applicantId(command.getApplicantId())
                .sealId(command.getSealId())
                .matterId(command.getMatterId())
                .documentId(command.getDocumentId())
                .applicationReason(command.getApplicationReason())
                .documentName(command.getDocumentName())
                .copies(command.getCopies())
                .usageDate(command.getUsageDate())
                .urgencyLevel(command.getUrgencyLevel())
                .status("PENDING")
                .build();
        
        sealApplicationRepository.save(application);
        
        // 4. 启动审批流程
        String processInstanceId = workflowService.startProcess(
                "seal_application_approval",
                application.getId().toString(),
                Map.of(
                        "applicationId", application.getId(),
                        "sealType", seal.getSealType(),
                        "urgencyLevel", application.getUrgencyLevel(),
                        "applicantId", application.getApplicantId()
                )
        );
        
        application.setProcessInstanceId(processInstanceId);
        sealApplicationRepository.updateById(application);
        
        return SealApplicationAssembler.toDTO(application);
    }
    
    /**
     * 审批用印申请
     */
    @Transactional
    public void approveApplication(Long applicationId, Long approverId, 
                                   Boolean approved, String comment) {
        SealApplication application = sealApplicationRepository.findById(applicationId)
                .orElseThrow(() -> new BusinessException("用印申请不存在"));
        
        if (!"PENDING".equals(application.getStatus())) {
            throw new BusinessException("申请已处理，不能重复审批");
        }
        
        if (approved) {
            application.setStatus("APPROVED");
            application.setApproverId(approverId);
            application.setApprovedAt(LocalDateTime.now());
            
            // 发送通知给印章保管员
            notificationService.sendSealApprovedNotification(application);
        } else {
            application.setStatus("REJECTED");
            application.setApproverId(approverId);
            application.setApprovedAt(LocalDateTime.now());
            application.setRejectionReason(comment);
            
            // 发送通知给申请人
            notificationService.sendSealRejectedNotification(application, comment);
        }
        
        sealApplicationRepository.updateById(application);
        
        // 完成工作流任务
        workflowService.completeTask(applicationId.toString(), "approve", 
                Map.of("approved", approved, "comment", comment));
    }
    
    /**
     * 执行用印
     */
    @Transactional
    public void executeSeal(Long applicationId, Long operatorId, String sealedFileUrl) {
        SealApplication application = sealApplicationRepository.findById(applicationId)
                .orElseThrow(() -> new BusinessException("用印申请不存在"));
        
        if (!"APPROVED".equals(application.getStatus())) {
            throw new BusinessException("只有已批准的申请可以执行用印");
        }
        
        // 验证操作员是印章保管员
        Seal seal = sealRepository.findById(application.getSealId())
                .orElseThrow(() -> new BusinessException("印章不存在"));
        
        if (!seal.getKeeperId().equals(operatorId)) {
            throw new BusinessException("只有印章保管员可以执行用印");
        }
        
        // 更新申请状态
        application.setStatus("USED");
        application.setOperatorId(operatorId);
        application.setOperatedAt(LocalDateTime.now());
        application.setSealedFileUrl(sealedFileUrl);
        
        sealApplicationRepository.updateById(application);
        
        // 更新印章状态（如果用印后需要归还）
        if (seal.isRequireReturn()) {
            seal.setStatus("IN_USE");
            sealRepository.updateById(seal);
        }
        
        // 发送通知给申请人
        notificationService.sendSealUsedNotification(application);
    }
    
    private String generateApplicationNo() {
        return "SA" + LocalDate.now().format(DateTimeFormatter.ofPattern("yyyyMMdd")) +
               String.format("%04d", getDailySequence());
    }
```

```java
     * 打卡
     */
    @Transactional
    public AttendanceDTO checkInOut(CheckInOutCommand command) {
        LocalDate today = LocalDate.now();
        
        // 查找今天的考勤记录
        Attendance attendance = attendanceRepository
                .findByUserIdAndDate(command.getUserId(), today)
                .orElseGet(() -> createNewAttendance(command.getUserId(), today));
        
        // 判断是上班打卡还是下班打卡
        LocalTime now = LocalTime.now();
        LocalTime noon = LocalTime.of(12, 0);
        
        if (now.isBefore(noon)) {
            // 上班打卡
            if (attendance.getCheckInTime() != null) {
                throw new BusinessException("今日已打过上班卡");
            }
            attendance.setCheckInTime(LocalDateTime.now());
        } else {
            // 下班打卡
            if (attendance.getCheckOutTime() != null) {
                throw new BusinessException("今日已打过下班卡");
            }
            if (attendance.getCheckInTime() == null) {
                throw new BusinessException("请先打上班卡");
            }
            attendance.setCheckOutTime(LocalDateTime.now());
            
            // 计算工时
            calculateWorkHours(attendance);
        }
        
        attendanceRepository.saveOrUpdate(attendance);
        return AttendanceAssembler.toDTO(attendance);
    }
    
    /**
     * 计算工时
     */
    private void calculateWorkHours(Attendance attendance) {
        if (attendance.getCheckInTime() == null || attendance.getCheckOutTime() == null) {
            return;
        }
        
        Duration duration = Duration.between(
                attendance.getCheckInTime(), 
                attendance.getCheckOutTime()
        );
        
        // 扣除午休时间（假设12:00-13:30为午休）
        LocalTime checkInTime = attendance.getCheckInTime().toLocalTime();
        LocalTime checkOutTime = attendance.getCheckOutTime().toLocalTime();
        LocalTime lunchStart = LocalTime.of(12, 0);
        LocalTime lunchEnd = LocalTime.of(13, 30);
        
        long lunchMinutes = 0;
        if (checkInTime.isBefore(lunchEnd) && checkOutTime.isAfter(lunchStart)) {
            LocalTime actualLunchStart = checkInTime.isBefore(lunchStart) ? lunchStart : checkInTime;
            LocalTime actualLunchEnd = checkOutTime.isAfter(lunchEnd) ? lunchEnd : checkOutTime;
            
            if (actualLunchStart.isBefore(actualLunchEnd)) {
                lunchMinutes = Duration.between(actualLunchStart, actualLunchEnd).toMinutes();
            }
        }
        
        long workMinutes = duration.toMinutes() - lunchMinutes;
        double workHours = workMinutes / 60.0;
        
        attendance.setWorkHours(BigDecimal.valueOf(workHours).setScale(2, RoundingMode.HALF_UP));
        
        // 判断是否异常
        checkAttendanceException(attendance);
    }
    
    /**
     * 检查考勤异常
     */
    private void checkAttendanceException(Attendance attendance) {
        // 上班时间：9:00
        LocalTime workStartTime = LocalTime.of(9, 0);
        // 下班时间：18:00
        LocalTime workEndTime = LocalTime.of(18, 0);
        
        LocalTime checkInTime = attendance.getCheckInTime().toLocalTime();
        LocalTime checkOutTime = attendance.getCheckOutTime().toLocalTime();
        
        // 迟到判断
        if (checkInTime.isAfter(workStartTime.plusMinutes(30))) { // 30分钟缓冲
            attendance.setStatus("LATE");
            attendance.setExceptionType("迟到");
            attendance.setExceptionReason("上班打卡时间晚于9:30");
        }
        // 早退判断
        else if (checkOutTime.isBefore(workEndTime.minusMinutes(30))) { // 30分钟缓冲
            attendance.setStatus("LEAVE_EARLY");
            attendance.setExceptionType("早退");
            attendance.setExceptionReason("下班打卡时间早于17:30");
        }
        // 正常
        else {
            attendance.setStatus("NORMAL");
        }
    }
    
    /**
     * 获取月度考勤统计
     */
    public AttendanceStatDTO getMonthlyStat(Long userId, int year, int month) {
        LocalDate startDate = LocalDate.of(year, month, 1);
        LocalDate endDate = startDate.withDayOfMonth(startDate.lengthOfMonth());
        
        List<Attendance> attendances = attendanceRepository
                .findByUserAndDateRange(userId, startDate, endDate);
        
        long normalDays = attendances.stream()
                .filter(a -> "NORMAL".equals(a.getStatus()))
                .count();
        
        long lateDays = attendances.stream()
                .filter(a -> "LATE".equals(a.getStatus()))
                .count();
        
        long leaveEarlyDays = attendances.stream()
                .filter(a -> "LEAVE_EARLY".equals(a.getStatus()))
                .count();
        
        long absentDays = startDate.lengthOfMonth() - attendances.size();
        
        BigDecimal totalHours = attendances.stream()
                .map(Attendance::getWorkHours)
                .filter(Objects::nonNull)
                .reduce(BigDecimal.ZERO, BigDecimal::add);
        
        return AttendanceStatDTO.builder()
                .userId(userId)
                .year(year)
                .month(month)
                .totalDays(startDate.lengthOfMonth())
                .normalDays(normalDays)
                .lateDays(lateDays)
                .leaveEarlyDays(leaveEarlyDays)
                .absentDays(absentDays)
                .totalWorkHours(totalHours)
                .averageDailyHours(totalHours.divide(
                        BigDecimal.valueOf(attendances.size()), 2, RoundingMode.HALF_UP))
                .build();
    }
}
```

```java
    @Transactional
    public KnowledgeArticleDTO create(CreateArticleCommand command) {
        // 验证分类存在
        if (!knowledgeCategoryRepository.existsById(command.getCategoryId())) {
            throw new BusinessException("知识分类不存在");
        }
        
        // 创建文章
        KnowledgeArticle article = KnowledgeArticle.builder()

                .summary(command.getSummary())
                .keywords(command.getKeywords())
                .attachments(command.getAttachments())
                .sourceType(command.getSourceType())
                .sourceMatterId(command.getSourceMatterId())
                .visibility(command.getVisibility())
                .status("DRAFT")
                .createdBy(command.getAuthorId())
                .build();
        
        knowledgeArticleRepository.save(article);
        
        return KnowledgeArticleAssembler.toDTO(article);
    }
    
    @Transactional
    public void publishArticle(Long articleId, Long publisherId) {
        KnowledgeArticle article = knowledgeArticleRepository.findById(articleId)
                .orElseThrow(() -> new BusinessException("文章不存在"));
        
        if (!"DRAFT".equals(article.getStatus())) {
            throw new BusinessException("只有草稿状态的文章可以发布");
        }
        
        // 验证发布权限
        if (!canPublishArticle(publisherId, article)) {
            throw new BusinessException("没有发布权限");
        }
        
        article.setStatus("PUBLISHED");
        article.setPublishedAt(LocalDateTime.now());
        article.setPublishedBy(publisherId);
        
        knowledgeArticleRepository.updateById(article);
        
        // 发送通知
        notificationService.sendArticlePublishedNotification(article);
    }
    
    public PageResult<KnowledgeArticleDTO> searchArticles(ArticleSearchRequest request) {
        Page<KnowledgeArticle> page = knowledgeArticleRepository.search(
                request.getPage(),
                request.getPageSize(),
                request.getKeyword(),
                request.getCategoryId(),
                request.getSourceType(),
                request.getVisibility(),
                request.getStatus(),
                request.getStartDate(),
                request.getEndDate()
        );
        
        List<KnowledgeArticleDTO> items = page.getRecords().stream()
                .map(KnowledgeArticleAssembler::toDTO)
                .collect(Collectors.toList());
        
        return PageResult.of(page.getTotal(), items, request.getPage(), request.getPageSize());
    }
    
    @Transactional
    public void increaseViewCount(Long articleId) {
        knowledgeArticleRepository.incrementViewCount(articleId);
    }
    
    @Transactional
    public void likeArticle(Long articleId, Long userId) {
        // 检查是否已经点赞
        if (articleLikeRepository.existsByArticleIdAndUserId(articleId, userId)) {
            throw new BusinessException("已经点过赞了");
        }
        
        // 记录点赞
        ArticleLike like = ArticleLike.builder()
                .articleId(articleId)
                .userId(userId)
                .build();
        
        articleLikeRepository.save(like);
        
        // 更新文章点赞数
        knowledgeArticleRepository.incrementLikeCount(articleId);
    }
    
    private boolean canPublishArticle(Long userId, KnowledgeArticle article) {
        // 作者可以发布自己的文章
        if (article.getCreatedBy().equals(userId)) {
            return true;
        }
        
        // 团队负责人可以发布团队文章
        if ("TEAM".equals(article.getVisibility())) {
            return isTeamLeader(userId, article.getCreatedBy());
        }
        
        // 管理员可以发布所有文章
        return userService.isAdmin(userId);
    }
    
    private String generateArticleNo() {
        return "KA" + LocalDate.now().format(DateTimeFormatter.ofPattern("yyyyMMdd")) +
               String.format("%04d", getDailySequence());
    }
```

     - 特别是涉及 `LEFT JOIN` 的复杂查询时，MyBatis 可能无法正确映射 `Long` 类型
   - **已尝试的解决方案**：
     - 尝试使用 `CAST(...AS BIGINT)` 和 `::bigint` 语法
     - 尝试添加 `@ResultType(Long.class)` 注解
     - 改用 MyBatis-Plus 的 `count()` 方法（部分有效，但仍有问题）

```java
// domain/finance/service/CommissionCalculator.java
package com.lawfirm.domain.finance.service;

import com.lawfirm.domain.finance.entity.Commission;
import com.lawfirm.domain.finance.entity.CommissionRule;
import com.lawfirm.domain.finance.entity.Payment;
import com.lawfirm.domain.matter.entity.Matter;
import com.lawfirm.domain.matter.entity.MatterParticipant;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.math.BigDecimal;
import java.math.RoundingMode;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;

@Slf4j
@Service
@RequiredArgsConstructor
public class CommissionCalculator {
    
    private final CommissionRuleRepository ruleRepository;
    
    /**
     * 计算提成
     * 
     * 业务规则：
     * 1. 净收入 = 到账金额 - 税费 - 管理费 - 办案成本
     * 2. 个人净收入 = 净收入 × 分配比例
     * 3. 提成 = 个人净收入 × 提成比例（阶梯计算）
     */
    public CommissionResult calculate(Payment payment, Matter matter, 
                                       List<MatterParticipant> participants,
                                       BigDecimal costAmount) {
        
        // 1. 获取适用的提成规则
        CommissionRule rule = findApplicableRule(matter);
        
        // 2. 计算净收入
        BigDecimal grossAmount = payment.getAmount();
        BigDecimal taxAmount = grossAmount.multiply(rule.getTaxRate())
                .setScale(2, RoundingMode.HALF_UP);
        BigDecimal managementFee = grossAmount.multiply(rule.getManagementFeeRate())
                .setScale(2, RoundingMode.HALF_UP);
        BigDecimal netAmount = grossAmount
                .subtract(taxAmount)
                .subtract(managementFee)
                .subtract(costAmount != null ? costAmount : BigDecimal.ZERO);
        
        log.info("提成计算 - 毛收入:{}, 税费:{}, 管理费:{}, 成本:{}, 净收入:{}", 
                grossAmount, taxAmount, managementFee, costAmount, netAmount);
        
        // 3. 计算每个参与人的提成
        List<Commission> commissions = new ArrayList<>();
        
        for (MatterParticipant participant : participants) {
            BigDecimal distributionRatio = participant.getDistributionRatio();
            
            // 该参与人的净收入份额
            BigDecimal participantNetAmount = netAmount.multiply(distributionRatio)
                    .setScale(2, RoundingMode.HALF_UP);
            
            // 根据阶梯计算提成比例
            BigDecimal commissionRate = calculateTieredRate(participantNetAmount, rule);
            
            // 提成金额
            BigDecimal commissionAmount = participantNetAmount.multiply(commissionRate)
                    .setScale(2, RoundingMode.HALF_UP);
            
            Commission commission = Commission.builder()
                    .paymentId(payment.getId())
                    .matterId(matter.getId())
                    .userId(participant.getUserId())
                    .ruleId(rule.getId())
                    .grossAmount(grossAmount.multiply(distributionRatio).setScale(2, RoundingMode.HALF_UP))
                    .taxAmount(taxAmount.multiply(distributionRatio).setScale(2, RoundingMode.HALF_UP))
                    .managementFee(managementFee.multiply(distributionRatio).setScale(2, RoundingMode.HALF_UP))
                    .costAmount(costAmount != null ? 
                            costAmount.multiply(distributionRatio).setScale(2, RoundingMode.HALF_UP) : BigDecimal.ZERO)
                    .netAmount(participantNetAmount)
                    .distributionRatio(distributionRatio)
                    .commissionRate(commissionRate)
                    .commissionAmount(commissionAmount)
                    .status("CALCULATED")
                    .calculationSnapshot(buildCalculationSnapshot(rule, participant, participantNetAmount, commissionRate))
                    .build();
            
            commissions.add(commission);
            
            log.info("参与人:{}, 分配比例:{}, 净收入:{}, 提成比例:{}, 提成金额:{}", 
                    participant.getUserId(), distributionRatio, participantNetAmount, 
                    commissionRate, commissionAmount);
        }
        
        return new CommissionResult(payment.getId(), matter.getId(), commissions);
    }
    
    /**
     * 阶梯提成比例计算
     */
    private BigDecimal calculateTieredRate(BigDecimal amount, CommissionRule rule) {
        List<RateTier> tiers = rule.getRateTiers();
        
        for (RateTier tier : tiers) {
            boolean aboveMin = amount.compareTo(tier.getMinAmount()) >= 0;
            boolean belowMax = tier.getMaxAmount() == null || 
                               amount.compareTo(tier.getMaxAmount()) < 0;
            
            if (aboveMin && belowMax) {
                return tier.getRate();
            }
        }
        
        // 默认返回最高档
        return tiers.get(tiers.size() - 1).getRate();
    }
    
    /**
     * 查找适用的提成规则
     */
    private CommissionRule findApplicableRule(Matter matter) {
        // 按优先级查找：个人规则 > 部门规则 > 业务类型规则 > 默认规则
        return ruleRepository.findApplicableRule(
                matter.getMainLawyerId(),
                matter.getDepartmentId(),
                matter.getBusinessType()
        ).orElseThrow(() -> new BusinessException("未找到适用的提成规则"));
    }
    
    private String buildCalculationSnapshot(CommissionRule rule, MatterParticipant participant,
                                           BigDecimal netAmount, BigDecimal rate) {
        Map<String, Object> snapshot = Map.of(
                "ruleId", rule.getId(),
                "ruleName", rule.getRuleName(),
                "taxRate", rule.getTaxRate(),
                "managementFeeRate", rule.getManagementFeeRate(),
                "participantRole", participant.getRole(),
                "distributionRatio", participant.getDistributionRatio(),
                "netAmount", netAmount,
                "commissionRate", rate,
                "calculatedAt", LocalDateTime.now()
        );
        return JsonUtils.toJson(snapshot);
    }
}
```

```java
        return pendingMatters.stream()
                .map(matter -> calculateMatchScore(matter, ocrResult))
                .filter(candidate -> candidate.getScore().compareTo(new BigDecimal("0.3")) > 0)
                .sorted(Comparator.comparing(MatchCandidate::getScore).reversed())
                .limit(10)
                .collect(Collectors.toList());
    }
    
    /**
     * 计算匹配分数
     */
    private MatchCandidate calculateMatchScore(Matter matter, OcrResult ocrResult) {
        BigDecimal score = BigDecimal.ZERO;
        List<String> matchReasons = new ArrayList<>();
        
        // 1. 金额匹配（权重40%）
        BigDecimal pendingAmount = contractRepository.getPendingAmount(matter.getId());
        if (pendingAmount != null && ocrResult.getAmount() != null) {
            BigDecimal amountDiff = pendingAmount.subtract(ocrResult.getAmount()).abs();
            BigDecimal amountRatio = amountDiff.divide(pendingAmount, 4, RoundingMode.HALF_UP);
            
            if (amountRatio.compareTo(new BigDecimal("0.01")) <= 0) {
                score = score.add(new BigDecimal("0.40"));
                matchReasons.add("金额完全匹配");
            } else if (amountRatio.compareTo(new BigDecimal("0.05")) <= 0) {
                score = score.add(new BigDecimal("0.30"));
                matchReasons.add("金额接近匹配");
            }
        }
        
        // 2. 付款方名称匹配（权重35%）
        Client client = clientRepository.findById(matter.getClientId()).orElse(null);
        if (client != null && ocrResult.getPayer() != null) {
            String clientName = client.getClientName();
            String payer = ocrResult.getPayer();
            
            if (clientName.equals(payer)) {
                score = score.add(new BigDecimal("0.35"));
                matchReasons.add("付款方名称完全匹配");
            } else if (clientName.contains(payer) || payer.contains(clientName)) {
                score = score.add(new BigDecimal("0.25"));
                matchReasons.add("付款方名称部分匹配");
            }
        }
        
        // 3. 备注信息匹配（权重25%）
        if (ocrResult.getRemark() != null) {
            String remark = ocrResult.getRemark().toLowerCase();
            if (remark.contains(matter.getMatterNo().toLowerCase()) ||
                remark.contains(matter.getMatterName().toLowerCase())) {
                score = score.add(new BigDecimal("0.25"));
                matchReasons.add("备注包含项目信息");
            }
        }
        
        return MatchCandidate.builder()
                .matter(MatterAssembler.toSimpleDTO(matter))
                .score(score)
                .reasons(matchReasons)
                .pendingAmount(pendingAmount)
                .build();
    }
    
    /**
     * 确认核销

    @Transactional
    public PaymentDTO confirmReconciliation(ConfirmReconciliationCommand command) {

        return PaymentAssembler.toDTO(payment);
    }

     * 财务确认收款（触发提成计算）

        payment.confirm(confirmerId);
        paymentRepository.updateById(payment);
        
        // 2. 计算提成

                .orElseThrow(() -> new BusinessException("项目不存在"));

        // 3. 保存提成记录
        commissionRepository.saveBatch(result.getCommissions());
        
        // 4. 更新合同收款进度
        contractRepository.updateReceivedAmount(payment.getContractId(), payment.getAmount());
        
        log.info("收款确认完成，生成{}条提成记录", result.getCommissions().size());
    }
```

```java
package com.lawfirm.application.workflow.service;

import com.lawfirm.domain.workflow.entity.WorkflowConfig;
import com.lawfirm.domain.system.repository.WorkflowConfigRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.cache.annotation.Cacheable;
import org.springframework.stereotype.Service;

import java.math.BigDecimal;
import java.util.Comparator;
import java.util.List;
import java.util.stream.Collectors;

@Service
@RequiredArgsConstructor
public class WorkflowConfigService {
    
    private final WorkflowConfigRepository configRepository;
    
    /**
     * 获取审批流程配置
     */
    @Cacheable(value = "workflowConfig", key = "#businessType + '_' + #conditionValue")
    public List<ApprovalNode> getApprovalNodes(String businessType, BigDecimal conditionValue) {
        List<WorkflowConfig> configs = configRepository.findByBusinessTypeAndActive(businessType, true);
        
        // 根据条件筛选适用的配置
        List<WorkflowConfig> applicableConfigs = configs.stream()
                .filter(config -> matchCondition(config, conditionValue))
                .sorted(Comparator.comparing(WorkflowConfig::getPriority).reversed()
                        .thenComparing(WorkflowConfig::getNodeOrder))
                .collect(Collectors.toList());
        
        // 转换为审批节点
        return applicableConfigs.stream()
                .map(this::toApprovalNode)
                .collect(Collectors.toList());
    }
    
    private boolean matchCondition(WorkflowConfig config, BigDecimal value) {
        if (config.getConditionType() == null || "NONE".equals(config.getConditionType())) {
            return true;
        }
        
        if (!"AMOUNT".equals(config.getConditionType())) {
            return true;
        }
        
        BigDecimal min = config.getConditionValueMin();
        BigDecimal max = config.getConditionValueMax();
        
        switch (config.getConditionOperator()) {
            case "LT":
                return value.compareTo(max) < 0;
            case "LTE":
                return value.compareTo(max) <= 0;
            case "GT":
                return value.compareTo(min) > 0;
            case "GTE":
                return value.compareTo(min) >= 0;
            case "BETWEEN":
                return value.compareTo(min) >= 0 && value.compareTo(max) < 0;
            default:
                return true;
        }
    }
    
    private ApprovalNode toApprovalNode(WorkflowConfig config) {
        return ApprovalNode.builder()
                .nodeOrder(config.getNodeOrder())
                .nodeName(config.getNodeName())
                .approverType(config.getApproverType())
                .approverRole(config.getApproverRole())
                .approverUserId(config.getApproverUserId())
                .build();
    }
```
